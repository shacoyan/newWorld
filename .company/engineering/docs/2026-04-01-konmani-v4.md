### B2: index.js — 出勤ボタンのロジック追加

index.js の bindEvents() 内に以下を追加:

```js
// 出勤ボタン
calendarGridEl.addEventListener('click', function(e) {
  if (!e.target.classList.contains('btn-checkin')) return;
  e.stopPropagation();
  var dateKey = e.target.dataset.key;
  var defaultStart = data.settings.defaultStartTime || '';
  if (!defaultStart) {
    alert('デフォルト出勤時刻が設定されていません。設定画面から登録してください。');
    return;
  }
  ensureRecord(data, dateKey).startTime = defaultStart;
  persistData(data);
  renderCalendar();
  // 今日のセルなら今日セクションも更新
  if (dateKey === todayKey) renderToday();
});
```

また、calendarGridEl の既存クリックイベントで `.btn-checkin` を除外する:
```js
calendarGridEl.addEventListener('click', function(e) {
  if (e.target.classList.contains('btn-checkin')) return; // 追加
  var cell = e.target.closest('.day-cell');
  ...
});
```

また、ユーザー名チップの初期設定:
auth.js が `app:ready` を発火するタイミングで `window.currentUser` が存在する想定。
index.js の renderAll() 内で:
```js
// ユーザー名設定
var displayName = (window.currentUser && (window.currentUser.displayName || window.currentUser.email)) || '';
var nameEl = document.getElementById('user-name');
var initialEl = document.getElementById('user-initial');
if (nameEl) nameEl.textContent = displayName;
if (initialEl) initialEl.textContent = displayName ? displayName.charAt(0).toUpperCase() : '?';
```

### C3: settings.js — 新設定のロジック追加

`init()` に追加:
```js
var defaultStartEl = document.getElementById('default-start-time');
var timeStepRadios = document.querySelectorAll('input[name="timeStep"]');

defaultStartEl.value = data.settings.defaultStartTime || '';
timeStepRadios.forEach(function(r) { r.checked = (parseInt(r.value) === (data.settings.timeStep || 1)); });
```

`bindEvents()` に追加:
```js
defaultStartEl.addEventListener('change', function() {
  data.settings.defaultStartTime = this.value;
  persistData(data);
});
timeStepRadios.forEach(function(r) {
  r.addEventListener('change', function() {
    data.settings.timeStep = parseInt(this.value);
    persistData(data);
  });
});
```

また `init()` の `toggleSalarySection` の初期呼び出しで、デフォルト 'hourly' の場合に hourly-section が表示されるよう確認。
